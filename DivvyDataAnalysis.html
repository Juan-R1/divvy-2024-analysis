# Create month column from ride start time
df['month'] = df['started_at'].dt.to_period('M').astype(str)

# Group ride counts by month and user type
monthly = df.groupby(['month', 'member_casual']).size().reset_index(name='ride_count')

# Line plot of monthly rides
plt.figure(figsize=(12,6))
sns.lineplot(data=monthly, x='month', y='ride_count', hue='member_casual', marker='o')
plt.title("Monthly Ride Trends (2024)")
plt.xlabel("Month")
plt.ylabel("Number of Rides")
